import { createDeliversMap, createConfigMap } from './utils';
import RunningSet from './RunningSet';
 
export const GLOBAL_DEPENDENCE_SCOPE = '*';
 
export default class Graph {
  config: GraphConfig[];
  runningSet: RunningSet;
  configMap: Map<string, GraphConfig>;
  deliverMap: Map<string, string[]>;
  constructor(config: GraphConfig[]) {
    this.config = config;
    this.configMap = createConfigMap(config);
    this.deliverMap = createDeliversMap(config);
    this.runningSet = new RunningSet();
  }
 
  /**
   *
   * @param config 找到所有起点, 规则: 没有依赖，或者依赖全都不在dfsConfig里面
   */
  public getFirstPoints(scope?: string) {
    const startPoints = [];
    // 通过作用于过滤点
    const config = this.getConfigByScope(scope);
    for (const item of config) {
      const deps = (item.deps || []).filter(point =>
        this.checkIsInnerPoints(point, scope)
      );
      if (scope) {
        console.log(22, deps);
      }
      if (deps.length === 0) {
        startPoints.push(item.key);
      }
    }
    return startPoints;
  }
 
  /**
   * 找到当前起点在作用于下面的所有点,过滤掉所有非内部的点
   * @param startKey 起点
   * @param scope 作用域
   */
  public getPointsByScope(startKey: string[], scope?: string) {
    const allPoints = this.getAllPointsByPoints(startKey, scope);
    console.log(222, allPoints);
    return this.filterInnerPoints(allPoints, scope);
  }
 
  /**
   * 过滤掉非内部的点
   * @param points
   */
  private filterInnerPoints(points, scope) {
    return points.filter(point => this.checkIsInnerPoints(point, scope));
  }
 
  /**
   * 检查当前点是否是图的内部点
   * @param point 点
   */
  private checkIsInnerPoints(point, scope?: string) {
    return this.getConfigByScope(scope).some(item => item.key === point);
  }
  /**
   * 找到当前点的所有起点
   * @param startKey
   * @param scope 作用域
   */
  public getStartPointByKey(
    startKeys: string[],
    current: string,
    scope?: string
  ) {
    const relationStartKeys = [] as string[];
    startKeys.forEach(startKey => {
      const allPointByStartKey = this.findPointsByScope([startKey], scope);
      if (allPointByStartKey.includes(current)) {
        relationStartKeys.push(current);
      }
    });
    return relationStartKeys;
  }
 
  /**
   * 获取当前作用域的依赖关系
   * @param scope 作用域
   */
  private getConfigByScope(scope?) {
    if (!scope || scope === GLOBAL_DEPENDENCE_SCOPE) {
      return this.config;
    }
    return this.config.filter(item => item.scope === scope);
  }
 
  /**
   * 找到当前点关联的所有点
   * @param points
   * @param createDeliversMap
   */
  private getAllPointsByPoints(points: string[], scope?: string) {
    // 创建派发map
    const deliverMap = createDeliversMap(this.getConfigByScope(scope));
    const allPointSet = new Set();
    // 建立当前需要执行的事件列表
    const whenChange = (key: string) => {
      allPointSet.add(key);
 
      const currentDependence = deliverMap.get(key) || [];
      // 寻找所有直接依赖以及间接依赖项
      for (const dependent of currentDependence) {
        // 添加任务
        whenChange(dependent);
      }
    };
    points.forEach(currentStartKey => {
      whenChange(currentStartKey);
    });
    return Array.from(allPointSet) as string[];
  }
}
